# apt packages for Streamlit Community Cloud (installed before pip/uv).
#
# ultralytics pulls in the GUI build of opencv-python, whose `import cv2` links
# two native libs the container does not ship: libGL.so.1 (from libgl1) and
# libgthread-2.0.so.0 (from libglib2.0-0).
#
# Community Cloud runs Debian trixie, where libglib2.0-0 was renamed to
# libglib2.0-0t64 (the 64-bit time_t transition). The bare name `libglib2.0-0`
# only resolves to an old bullseye package whose libffi7/libpcre3 dependencies
# are not installable on trixie, so apt fails the whole file. Use the t64 name.
libgl1
libglib2.0-0t64
